function generateHashtag(str) {
  str = str.trim();
  if (str.length === 0) return false;
  const words = str.split(/\s+/g);
  const finalWord = "#" + words.map((word) => word[0].toUpperCase() + word.substring(1)).join("");
  return finalWord.length > 140 ? false : finalWord;
}
